Description

Basic operations on type-level Booleans.

Since: 4.7.0.0

type family If cond tru fls Source

Type-level If. If True a b ==> a; If False a b ==> b

Equations

If True tru fls = tru
If False tru fls = fls

type family a && b infixr 3 Source

Type-level "and"

Equations

False && a = False
True && a = a
a && False = False
a && True = a
a && a = a

type family a || b infixr 2 Source

Type-level "or"

Equations

False || a = a
True || a = True
a || False = a
a || True = True
a || a = a

type family Not a Source

Type-level "not"

Equations

Not False = True
Not True = False
